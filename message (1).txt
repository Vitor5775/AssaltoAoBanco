
dinheiro_pessoas = []
for i in range(10):
    valor = int(input(f"Digite a quantidade de dinheiro da pessoa {i+1}: "))
    dinheiro_pessoas.append(valor)
dinheiro_bandido_confesso = int(input("Digite a quantidade de dinheiro que o bandido confesso revelou: "))

total_roubado = 0
for valor in dinheiro_pessoas:
    if valor % 10 == 0 and valor >= dinheiro_bandido_confesso:
        total_roubado += valor
print(f"O total roubado foi: {total_roubado} reais")
